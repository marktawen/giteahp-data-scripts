SELECT Country , COUNT(*) AS count
FROM (
SELECT hrm.Request , hrm."Source" , asgi.Country 
FROM http_requests_mitmproxy hrm 
LEFT JOIN all_sources_geo_info asgi ON hrm."Source" = asgi.ip_address
)
GROUP BY Country
ORDER BY count DESC
