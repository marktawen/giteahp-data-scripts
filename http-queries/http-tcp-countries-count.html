SELECT Country , COUNT(*) AS count
FROM (
SELECT htp.tcp_id , htp."Source" , asgi.Country 
FROM http_tcp_pcap htp 
LEFT JOIN all_sources_geo_info asgi ON htp."Source" = asgi.ip_address
)
GROUP BY Country
ORDER BY count DESC
