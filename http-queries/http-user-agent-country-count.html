FROM (
SELECT asgi.Country , hrm."User Agent"
FROM http_requests_mitmproxy hrm 
JOIN all_sources_geo_info asgi ON hrm."Source" = asgi.ip_address
ORDER BY Country
)
Group by Country, "User Agent"
