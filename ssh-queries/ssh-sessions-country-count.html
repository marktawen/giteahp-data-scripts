SELECT Country , COUNT(*) AS count
FROM (
SELECT sss.sessionid , sss.client_ip, asgi.Country 
FROM ssh_sessions_sshmitm sss
JOIN all_sources_geo_info asgi ON sss.client_ip = asgi.ip_address
group by sss.sessionid 
)
GROUP BY Country
ORDER BY count DESC
