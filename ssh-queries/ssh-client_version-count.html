FROM (
SELECT sss.client_version, sss.client_ip 
FROM ssh_sessions_sshmitm sss
where module= 'clientaudit'
GROUP BY sss.sessionid 
order by client_version
)
GROUP BY client_version 
ORDER BY count DESC
