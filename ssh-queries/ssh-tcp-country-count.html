SELECT Country , COUNT(*) AS count
FROM (
SELECT stp.tcp_id , stp."Source" , asgi.Country 
FROM ssh_tcp_pcap stp 
LEFT JOIN all_sources_geo_info asgi ON stp."Source" = asgi.ip_address
)
GROUP BY Country
ORDER BY count DESC
